Implement cross-kind name collision detection that fails at template rendering time when truncated resource names would collide between different deployments, CronJobs, or hooks.

Purpose: Prevents silent name collisions where two different resources get the same Kubernetes name after truncation, causing one to overwrite the other at apply time.
Output: A validation helper + template that checks all generated resource names and fails with an actionable error on collision.

    - deployment `deploy-aaaaaaaaa-2` + cronJob `cron-abcdefgh` = suffix `deploy-aaaaaaaaa-2-cron-abcdefgh` (31 chars, truncated to same 26)

    Full name pattern: `{prefix}{deploymentName}-{cronJobName}` truncated to 52.
    Prefix is 26 chars. So `{deploymentName}-{cronJobName}` gets 26 chars before truncation.
    - deployment `deploy-aaaa-1` + cronJob `cron-abcde1` = `deploy-aaaa-1-cron-abcde1` (25 chars, fits in 26)

    Append to the end of `charts/global-chart/templates/_helpers.tpl`:

    ```yaml
    {{/*
    Validate that all generated resource names are unique after truncation.
    Checks across deployments, CronJobs, hooks, and their sub-resources
    (including hook prerequisite ConfigMaps/Secrets and hook ServiceAccounts).
    Called from _validate.yaml.
    */}}
    {{- define "global-chart.validateNameCollisions" -}}
    {{- $root := . -}}
    {{- $fullname := include "global-chart.fullname" $root -}}
    {{- $names := dict -}}

    {{- /* 1. Deployment resource names (trunc 63) */ -}}
    {{- range $name, $deploy := .Values.deployments -}}
      {{- if $deploy -}}
      {{- if eq (include "global-chart.deploymentEnabled" (dict "deploy" $deploy)) "true" -}}
        {{- $depFullname := include "global-chart.deploymentFullname" (dict "root" $root "deploymentName" $name) -}}
        {{- if hasKey $names $depFullname -}}
          {{- fail (printf "Name collision: '%s' generated by deployment '%s' conflicts with %s" $depFullname $name (index $names $depFullname)) -}}
        {{- end -}}
        {{- $_ := set $names $depFullname (printf "deployment '%s'" $name) -}}

        {{- /* 1a. Deployment-level CronJob names (trunc 52) */ -}}
        {{- range $jobName, $job := $deploy.cronJobs -}}
          {{- if $job -}}
            {{- $jobFullname := printf "%s-%s" $depFullname $jobName | trunc 52 | trimSuffix "-" -}}
            {{- if hasKey $names $jobFullname -}}
              {{- fail (printf "Name collision: '%s' generated by cronJob '%s' in deployment '%s' conflicts with %s" $jobFullname $jobName $name (index $names $jobFullname)) -}}
            {{- end -}}
            {{- $_ := set $names $jobFullname (printf "cronJob '%s' in deployment '%s'" $jobName $name) -}}
          {{- end -}}
        {{- end -}}

        {{- /* 1b. Deployment-level Hook names (trunc 63) + hook SA + prerequisite ConfigMap/Secret */ -}}
        {{- if $deploy.hooks -}}
          {{- /* Hook prerequisite ConfigMap (trunc 63) — rendered once per deployment by Plan 01-01 */ -}}
          {{- $hasDeployConfigMap := and $deploy.configMap (gt (len $deploy.configMap) 0) -}}
          {{- if $hasDeployConfigMap -}}
            {{- $hookConfigName := printf "%s-hook-config" $depFullname | trunc 63 | trimSuffix "-" -}}
            {{- if hasKey $names $hookConfigName -}}
              {{- fail (printf "Name collision: '%s' generated by hook prerequisite ConfigMap for deployment '%s' conflicts with %s" $hookConfigName $name (index $names $hookConfigName)) -}}
            {{- end -}}
            {{- $_ := set $names $hookConfigName (printf "hook prerequisite ConfigMap for deployment '%s'" $name) -}}
          {{- end -}}

          {{- /* Hook prerequisite Secret (trunc 63) — rendered once per deployment by Plan 01-01 */ -}}
          {{- $hasDeploySecret := and $deploy.secret (gt (len $deploy.secret) 0) -}}
          {{- if $hasDeploySecret -}}
            {{- $hookSecretName := printf "%s-hook-secret" $depFullname | trunc 63 | trimSuffix "-" -}}
            {{- if hasKey $names $hookSecretName -}}
              {{- fail (printf "Name collision: '%s' generated by hook prerequisite Secret for deployment '%s' conflicts with %s" $hookSecretName $name (index $names $hookSecretName)) -}}
            {{- end -}}
            {{- $_ := set $names $hookSecretName (printf "hook prerequisite Secret for deployment '%s'" $name) -}}
          {{- end -}}

          {{- range $hookType, $jobs := $deploy.hooks -}}
            {{- range $jobName, $command := $jobs -}}
              {{- if $command -}}
                {{- $hookFullname := printf "%s-%s-%s" $depFullname $hookType $jobName | trunc 63 | trimSuffix "-" -}}
                {{- if hasKey $names $hookFullname -}}
                  {{- fail (printf "Name collision: '%s' generated by hook '%s/%s' in deployment '%s' conflicts with %s" $hookFullname $hookType $jobName $name (index $names $hookFullname)) -}}
                {{- end -}}
                {{- $_ := set $names $hookFullname (printf "hook '%s/%s' in deployment '%s'" $hookType $jobName $name) -}}

                {{- /* Hook per-command SA (if created) */ -}}
                {{- $deploySA := default (dict) $deploy.serviceAccount -}}
                {{- $commandSAMap := (and (hasKey $command "serviceAccount") (kindIs "map" $command.serviceAccount)) | ternary $command.serviceAccount (dict) -}}
                {{- $commandSAExplicitName := coalesce $command.serviceAccountName $commandSAMap.name -}}
                {{- $deploymentSAName := "" -}}
                {{- if $deploySA.create -}}
                  {{- $deploymentSAName = include "global-chart.deploymentServiceAccountName" (dict "root" $root "deploymentName" $name "deployment" $deploy) -}}
                {{- else if $deploySA.name -}}
                  {{- $deploymentSAName = $deploySA.name -}}
                {{- end -}}
                {{- $saCreate := false -}}
                {{- if not $commandSAExplicitName -}}
                  {{- if not $deploymentSAName -}}
                    {{- $saCreate = true -}}
                  {{- end -}}
                {{- else if and (hasKey $commandSAMap "create") $commandSAMap.create -}}
                  {{- $saCreate = true -}}
                {{- end -}}
                {{- if $saCreate -}}
                  {{- $hookSAName := $hookFullname -}}
                  {{- if hasKey $names $hookSAName -}}
                    {{- /* SA name equals hook job name — already tracked, skip duplicate check */ -}}
                  {{- end -}}
                {{- end -}}
              {{- end -}}
            {{- end -}}
          {{- end -}}
        {{- end -}}

      {{- end -}}
      {{- end -}}
    {{- end -}}

    {{- /* 2. Root-level CronJob names (trunc 52) */ -}}
    {{- range $name, $job := .Values.cronJobs -}}
      {{- if $job -}}
        {{- $jobFullname := printf "%s-%s" $fullname $name | trunc 52 | trimSuffix "-" -}}
        {{- if hasKey $names $jobFullname -}}
          {{- fail (printf "Name collision: '%s' generated by root cronJob '%s' conflicts with %s" $jobFullname $name (index $names $jobFullname)) -}}
        {{- end -}}
        {{- $_ := set $names $jobFullname (printf "root cronJob '%s'" $name) -}}
      {{- end -}}
    {{- end -}}

    {{- /* 3. Root-level Hook names (trunc 63) */ -}}
    {{- range $hookType, $jobs := .Values.hooks -}}
      {{- range $name, $command := $jobs -}}
        {{- if $command -}}
          {{- $hookFullname := include "global-chart.hookfullname" (merge (dict "hookname" $hookType "jobname" $name) $root) -}}
          {{- if hasKey $names $hookFullname -}}
            {{- fail (printf "Name collision: '%s' generated by root hook '%s/%s' conflicts with %s" $hookFullname $hookType $name (index $names $hookFullname)) -}}
          {{- end -}}
          {{- $_ := set $names $hookFullname (printf "root hook '%s/%s'" $hookType $name) -}}
        {{- end -}}
      {{- end -}}
    {{- end -}}

    {{- end }}
    ```
